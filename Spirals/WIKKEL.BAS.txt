10 REM ***WIKKELLIJN VAN CIRKEL***
20 REM ***NAAM:WIKKEL***
30 SCREEN 2 : CLS : PI=3.14159265
40 WINDOW (-12,-9)-(12,9)
45 A=1 : REM ***KEUZE STRAAL CIRKEL***
50 CIRCLE (0,0),A : PSET (A,0)
60 FOR N=0 TO 100 : T=2*PI*N/80
70 X=A*(COS(T)+T*SIN(T))
80 Y=A*(SIN(T)-T*COS(T))
90 LINE -(X,Y)
100 IF N MOD 10 = 0 THEN LINE (A*COS(T),A*SIN(T))-(X,Y)
110 NEXT N : PAINT (0,0) : END
